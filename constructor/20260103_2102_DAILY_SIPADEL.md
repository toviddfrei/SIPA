```python
import os
import logging
from core.config import (
    CRITICAL_DIRS,
    STRUCTURE_DIRS,
    CRITICAL_FILES,
    NAME_LOGGER_ESTRUCTURA
)

# Definición de la excepción de error crítico
class InfrastructureError(Exception):
    """Excepción para errores críticos de infraestructura."""
    pass

class EnvironmentManager:
    def __init__(self):
        # Usamos el nombre de logger definido en el gran configurador
        self.logger = logging.getLogger(NAME_LOGGER_ESTRUCTURA)

    def _check_write_permission(self, directory):
        """Verifica si el directorio tiene permisos de escritura."""
        if not os.access(directory, os.W_OK):
            error_msg = f"ERROR CRÍTICO: El directorio '{directory}' no tiene permisos de escritura."
            self.logger.critical(error_msg)
            raise InfrastructureError(error_msg)
        self.logger.debug(f"Permisos de escritura OK en: '{directory}'.")

    def check_all(self):
        """Ejecuta la auditoría completa de infraestructura."""
        try:
            self.logger.info("SIPAbap: Iniciando auditoría de entorno...")
            
            # 1. Validar y Crear Directorios Críticos (data, logs, db)
            for directory in CRITICAL_DIRS:
                if not os.path.exists(directory):
                    os.makedirs(directory, exist_ok=True)
                    self.logger.info(f"Directorio creado: {directory}")
                self._check_write_permission(directory)

            # 2. Validar Directorios de Estructura (core, interface, etc.)
            for directory in STRUCTURE_DIRS:
                if not os.path.exists(directory):
                    raise InfrastructureError(f"Directorio de estructura faltante: {directory}")
            
            # 3. Validar Ficheros Vitales (.sipa_root, identity.json, db)
            for file_path in CRITICAL_FILES:
                if not os.path.exists(file_path):
                    raise InfrastructureError(f"Fichero vital ausente: {file_path}")
            
            self.logger.info("SIPAbap: Infraestructura certificada [OK].")
            return True

        except Exception as e:
            self.logger.error(f"Fallo en la validación: {e}")
            return False
```
