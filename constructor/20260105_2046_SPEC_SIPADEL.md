1. Resolución: El fallo de la Shell (Rutas Absolutas)
El error No such file or directory ocurre porque la Shell no sabe dónde está parado el script. Vamos a estandarizar la llamada en main_dashboard.py.

Modifica el método run_db_test así:

Python

    def run_db_test(self):
        """Ejecución profesional con rutas absolutas."""
        import os
        # Localizamos la raíz del proyecto SIPA_PROJECT
        base_dir = os.path.dirname(os.path.dirname(os.path.abspath(__file__)))
        script_path = os.path.join(base_dir, "test_db_full.py")

        try:
            # Forzamos la ejecución en el directorio raíz del proyecto
            res = subprocess.run(['python3', script_path], 
                                 capture_output=True, text=True, cwd=base_dir)
            self.console_audit.config(state=tk.NORMAL)
            self.console_audit.delete(1.0, tk.END)
            self.console_audit.insert(tk.END, res.stdout + res.stderr)
            self.console_audit.config(state=tk.DISABLED)
        except Exception as e:
            tk.messagebox.showerror("Error SIPA", f"Fallo al localizar script: {e}")

    def listar_archivos_knowledge(self):
        """Escanea la carpeta de conocimiento y llena la tabla."""
        import os
        path = "data/knowledge"
        for item in self.tabla_docs.get_children(): self.tabla_docs.delete(item)
        
        if os.path.exists(path):
            for f in os.listdir(path):
                if f.endswith(".md"):
                    self.tabla_docs.insert("", tk.END, values=("Markdown", f))

    def editar_documento(self):
        """Abre el editor para el archivo seleccionado."""
        selected = self.tabla_docs.selection()
        if not selected: return
        
        filename = self.tabla_docs.item(selected)['values'][1]
        filepath = f"data/knowledge/{filename}"
        
        # Ventana de edición
        edit_win = tk.Toplevel(self.window)
        edit_win.title(f"Editando: {filename}")
        edit_win.geometry("800x600")
        
        txt_edit = tk.Text(edit_win, font=("Consolas", 10), undo=True)
        txt_edit.pack(fill=tk.BOTH, expand=True)
        
        with open(filepath, 'r') as f:
            txt_edit.insert(tk.END, f.read())
            
        def guardar():
            with open(filepath, 'w') as f:
                f.write(txt_edit.get(1.0, tk.END).strip())
            tk.messagebox.showinfo("SIPAdoc", "Archivo guardado con éxito.")
            
        tk.Button(edit_win, text="GUARDAR CAMBIOS", command=guardar, bg="#4CAF50", fg="white").pack(pady=5)
